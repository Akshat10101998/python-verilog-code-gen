module counter(clk,reset,select,counter);  

  

  parameter n = {{result.bits}};

  parameter start_point= {{result.spoint}};

  parameter end_point= {{result.epoint}};

  input clk;

  input reset;

  input[1:0] select;

  

  output reg[{{(result.bits | int)-1}}:0] counter;

  

  

  

  always @(posedge clk,reset)

    begin

      if (reset)

        counter <=start_point; 

      else begin

        if (counter==end_point)

                    counter<=start_point;

                else

                      counter =counter + 1;

        

      end

    end

endmodule
