Ring Counter

module ringcounter(Q,clk,reset);

  parameter n={{result.bits}};

  input clk,reset;

  output reg [n-1:0] Q;

  

  always @(posedge clk)

    if (reset)

      Q <= 1;

    else begin

      Q[n-1]<=Q[0];

      for (int i=0; i<n-1; i=i+1) begin

        Q[i] <= Q[i+1];

      end

    end

endmodule
