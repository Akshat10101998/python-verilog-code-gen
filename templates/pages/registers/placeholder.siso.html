module SISO_register(output_seq,input_seq,clk,reset);

  parameter n={{result.bits}};

  input [n-1:0] input_seq;

  input clk,reset;

  output reg [n-1:0] output_seq;

  integer i,j=0;

  wire input_temp;

  always @ (posedge clk) begin

    if (reset)

      output_seq <= 0;

    else 

      begin

        if (j<n) begin

          output_seq <= {output_seq[n-2:0],input_seq[j]};

          j = j+1;

        end

           else

          output_seq <= {output_seq[n-2:0],1'b0};

      end

  end

                 

endmodule
