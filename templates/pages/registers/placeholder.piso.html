  parameter n={{result.bits}};

  input clk,reset;

  input [n-1:0] input_seq;

  output [n-1:0] output_seq;

  wire [n-1:0] output_seq;

  reg [n-1:0]temp;

  

  always@(posedge clk,posedge reset)

    begin

      if(reset==1'b1)

        temp <= {{result.bits}}'b{% for i in range(result.bits | int) %}0{% endfor %};
