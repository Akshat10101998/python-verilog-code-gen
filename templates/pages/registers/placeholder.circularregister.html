Left Circular Register

module Circular_register(input_seq,clear,reset,clk,output_seq);

  parameter n = {{(result.bits | int)}};

  input [n-1:0] input_seq;

  input reset,clk,clear;

  output reg [n-1:0] output_seq;

  

  always @(posedge clk) begin

    if (clear)

      output_seq <= 0;

    if (reset)

      output_seq <= input_seq;

      else begin

      output_seq <= {output_seq[n-2:0],output_seq[n-1]};

    end

  end

  

endmodule
