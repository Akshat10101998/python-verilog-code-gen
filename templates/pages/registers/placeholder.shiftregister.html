Left Shift Register

module shiftregister(D,load,w,clk,Q);

  parameter n = {{result.bits}};

  input [n-1:0] D;

  input load,w,clk;

  output reg [n-1:0] Q;

  integer i,k;

  

  

  always @(posedge clk)

    if (load)

      Q <= D;

      else begin

      Q[0]<=w;

      for (i=1;i<n;i=i+1)

        Q[i]<=Q[i-1];

    end

endmodule

module shiftregister(D,load,w,clk,R);

  parameter n = {{result.bits}};

        R[k]<=R[k+1];

    end
