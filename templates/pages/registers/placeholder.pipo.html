PIPO

Module

module PIPO_register(input_seq, output_seq,reset, clk,clear);

  parameter n={{result.bits}};

  input [n-1:0] input_seq;

  input reset,clk,clear;

  output reg [n-1:0] output_seq; 

  

  always @ (posedge clk)

    begin

      if (clear)

        output_seq <= 0;

      if(reset)

        output_seq <= input_seq;

      else

        output_seq = {output_seq[n-2:0],1'b0};

    end

  

endmodule
