T Flip Flop

module Tff(reset,clk,t,q,qnot);

  input reset,clk,t;

  output reg q;

  output qnot;

  

  assign qnot = ~q;

  

  always @(posedge clk)

    begin

    if (reset)

      q <= 0;

    else

      if (t==0)

        q<=q;

       else if (t==1)

        q<=~q;

    end

  

endmodule
