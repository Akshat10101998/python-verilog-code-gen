SR Flip Flop

module SRff(reset,clk,s,r,q,qnot);

  input reset,clk,s,r;

  output reg q;

  output qnot;

  

  assign qnot = ~q;

  

  always @(posedge clk)

    begin

    if (reset)

      q <= 0;

    else

      if (s==0 & r==0)

        q<=q;

      else if (s==0 & r==1)

        q<=0;

      else if (s==1 & r==0)

        q<=1;

      else if (s==1 & r==1)

        q<=1'bx;

    end

  

endmodule
